        document.getElementById('authForm').addEventListener('submit', function(e) {
            e.preventDefault();
            const login = document.getElementById('login').value;
            const password = document.getElementById('password').value;
            const isSuccessful = mockAuthRequest(login, password);
            
            const messageDiv = document.getElementById('message');
            
            if (isSuccessful) {
                messageDiv.textContent = 'Авторизация выполнена';
            } else {
                messageDiv.textContent = 'Неверный логин или пароль';
            }
        });
        function mockAuthRequest(login, password) {
            return login.trim() !== '' && password.trim() !== '';
        }
    </script>

<!DOCTYPE html>
<html lang="ru">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>1</title>
</head>
<body>
    <div id="usersContainer"></div>
    <script>
        const mockUsers = [
            { name: "Какой-то Тип", email: "one@gmail.com" },
            { name: "Ктото", email: "gift@gmail.com" },
            { name: "VLone", email: "vl@gmail.com" }
        ];
        function loadUsers() {
            displayUsers(mockUsers);
        }
        function displayUsers(users) {
            const container = document.getElementById('usersContainer');
            container.innerHTML = '';
            if (users.length === 0) {
                container.textContent = 'Пользователи не найдены';
                return;
            }
            const list = document.createElement('ul');
            users.forEach(user => {
                const item = document.createElement('li');
                item.textContent = `${user.name} - ${user.email}`;
                list.appendChild(item);
            });
            container.appendChild(list);
        }
        window.addEventListener('DOMContentLoaded', loadUsers);
    </script>
</body>
</html>
